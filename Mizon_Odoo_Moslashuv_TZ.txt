class MizonLead(models.Model):
    _inherit = 'crm.lead'   # crm_lead jadvalini kengaytiradi
 
    # ── MANBA KANALI ──────────────────────────────────────────
    mizon_source = fields.Selection([
        ('meta_fb_ads',   'Facebook Lead Ads'),
        ('meta_fb_msg',   'Facebook Messenger'),
        ('meta_ig_dm',    'Instagram DM'),
        ('telegram_bot',  'Telegram Bot'),
        ('telegram_ch',   'Telegram Kanal'),
        ('phone_call',    'Telefon qo'ng'irog'i'),
        ('referral',      'Tavsiya (Referral)'),
        ('website',       'Veb-sayt'),
        ('manual',        'Qo'lda kiritilgan'),
    ], string='Mizon manbasi', default='manual',
       tracking=True)  # tracking=True => o'zgarish loglanadi
 
    # ── IJTIMOIY TARMOQ MA'LUMOTLARI ─────────────────────────
    telegram_username  = fields.Char('Telegram username', help='@username formatida')
    telegram_chat_id   = fields.Char('Telegram Chat ID', help='Bot suhbat ID')
    instagram_username = fields.Char('Instagram username')
    facebook_psid      = fields.Char('Facebook PSID', help='Page-Scoped User ID')
    meta_ad_id         = fields.Char('Meta Ad ID', help='Qaysi reklama orqali kelgan')
    meta_campaign_id   = fields.Char('Meta Campaign ID')
    meta_form_id       = fields.Char('Meta Lead Form ID')
 
    # ── BIZNES MA'LUMOTLARI ──────────────────────────────────
    company_type_mizon = fields.Selection([
        ('individual', 'Jismoniy shaxs'),
        ('small_biz',  'Kichik biznes (1-10 xodim)'),
        ('medium_biz', 'O'rta biznes (11-100 xodim)'),
        ('large_biz',  'Yirik korxona (100+ xodim)'),
    ], string='Kompaniya turi')
 
    product_interest   = fields.Many2many(
        'mizon.product.category',
        string='Qiziqish bildirgan mahsulot/xizmat'
    )
    budget_range = fields.Selection([
        ('under_1m',  '1 000 000 so'mgacha'),
        ('1m_5m',     '1-5 million so'm'),
        ('5m_20m',    '5-20 million so'm'),
        ('over_20m',  '20 million so'mdan yuqori'),
    ], string='Taxminiy byudjet')
 
    # ── LEAD SIFAT BALL ──────────────────────────────────────
    lead_score = fields.Integer(
        'Lead balli', default=0, readonly=True,
        help='Avtomatik hisoblanadi: 0-100'
    )
 
    # ── DUPLIKAT TEKSHIRISH ──────────────────────────────────
    is_duplicate    = fields.Boolean('Duplikat?', default=False, readonly=True)
    duplicate_lead_id = fields.Many2one('crm.lead', 'Asl lead (duplikat bo'lsa)')
 
    # ── AVTOMATIK LEAD SCORE HISOBLASH ───────────────────────
    @api.onchange('mizon_source', 'phone', 'email_from', 'partner_id')
    def _compute_lead_score(self):
        for rec in self:
            score = 0
            # Manba bo'yicha ball
            source_scores = {
                'meta_fb_ads': 30, 'meta_fb_msg': 25,
                'meta_ig_dm': 25,  'telegram_bot': 20,
                'telegram_ch': 20, 'phone_call': 35,
                'referral': 40,    'website': 15,
                'manual': 10,
            }
            score += source_scores.get(rec.mizon_source, 0)
            if rec.phone:       score += 10
            if rec.email_from:  score += 10
            if rec.partner_id:  score += 15
            rec.lead_score = min(score, 100)
 
    # ── DUPLIKAT TEKSHIRISH ───────────────────────────────────
    @api.model
    def check_duplicate(self, phone=None, email=None):
        domain = []
        if phone:
            domain.append(('phone', '=', phone))
        if email:
            domain.append(('email_from', '=', email))
        if not domain:
            return False
        existing = self.search(['|'] + domain, limit=1)
        return existing.id if existing else False

5.1.1 Odoo'ning controller tizimi
Odoo'da tashqi so'rovlarni qabul qilish uchun http.Controller ishlatiladi. Bu Django'dagi view'ga o'xshash:

# mizon_crm/controllers/meta_webhook.py
import json, hmac, hashlib
from odoo import http
from odoo.http import request
 
class MetaWebhookController(http.Controller):
 
    # ── WEBHOOK VERIFICATION (Meta talab qiladi) ──────────────
    @http.route('/mizon/webhook/meta', type='http',
                auth='none', methods=['GET'], csrf=False)
    def meta_verify(self, **kwargs):
        mode      = kwargs.get('hub.mode')
        token     = kwargs.get('hub.verify_token')
        challenge = kwargs.get('hub.challenge')
 
        # Sozlamalardagi token bilan taqqoslash
        config_token = request.env['ir.config_parameter'].sudo()\
            .get_param('mizon_crm.meta_verify_token')
 
        if mode == 'subscribe' and token == config_token:
            return request.make_response(challenge)
        return request.make_response('Forbidden', status=403)
 
    # ── WEBHOOK QABUL QILISH ──────────────────────────────────
    @http.route('/mizon/webhook/meta', type='http',
                auth='none', methods=['POST'], csrf=False)
    def meta_receive(self, **kwargs):
        payload = request.httprequest.data
        sig_header = request.httprequest.headers.get('X-Hub-Signature-256', '')
 
        # Imzoni tekshirish
        app_secret = request.env['ir.config_parameter'].sudo()\
            .get_param('mizon_crm.meta_app_secret')
        expected = 'sha256=' + hmac.new(
            app_secret.encode(), payload, hashlib.sha256
        ).hexdigest()
 
        if not hmac.compare_digest(sig_header, expected):
            return request.make_response('Invalid signature', status=403)
 
        data = json.loads(payload)
        # Lead Ads eventlarini qayta ishlash
        for entry in data.get('entry', []):
            for change in entry.get('changes', []):
                if change.get('field') == 'leadgen':
                    self._process_lead_ads(change['value'])
 
        return request.make_response('OK')
 
    def _process_lead_ads(self, lead_data):
        Lead = request.env['crm.lead'].sudo()
        vals = {
            'name': f"Meta Lead: {lead_data.get('form_id', '')}",
            'mizon_source': 'meta_fb_ads',
            'meta_form_id': lead_data.get('form_id'),
            'meta_ad_id': lead_data.get('ad_id'),
            'meta_campaign_id': lead_data.get('campaign_id'),
        }
        # Field values (forma maydonlari)
        for field in lead_data.get('field_data', []):
            if field['name'] == 'phone_number':
                vals['phone'] = field['values'][0]
            elif field['name'] == 'email':
                vals['email_from'] = field['values'][0]
            elif field['name'] == 'full_name':
                vals['contact_name'] = field['values'][0]
                vals['name'] = f"Meta Lead: {field['values'][0]}"
 
        lead = Lead.create(vals)
        lead._compute_lead_score()  # Ballni hisoblash

5.2.1 Telegram Webhook Controller
# mizon_crm/controllers/telegram_webhook.py
import json
from odoo import http
from odoo.http import request
 
class TelegramWebhookController(http.Controller):
 
    @http.route('/mizon/webhook/telegram', type='json',
                auth='none', methods=['POST'], csrf=False)
    def telegram_receive(self, **kwargs):
        data = json.loads(request.httprequest.data)
 
        # Xabar yoki Contact share
        message = data.get('message', {})
        chat_id = str(message.get('chat', {}).get('id', ''))
        text    = message.get('text', '')
        contact = message.get('contact')  # Telefon raqamini ulashganda
 
        # /start buyrug'i — menejerga ulash
        if text == '/start':
            self._link_manager(chat_id, message)
 
        # Kontakt ulashildi — lead yaratish
        elif contact:
            self._create_lead_from_contact(chat_id, contact)
 
        # Oddiy xabar — mavjud leadga biriktirish
        else:
            self._log_message(chat_id, text, message)
 
        return {'ok': True}
 
    def _create_lead_from_contact(self, chat_id, contact):
        Lead = request.env['crm.lead'].sudo()
        # Duplikat tekshirish
        existing = Lead.check_duplicate(phone=contact.get('phone_number'))
        if existing:
            # Mavjud leadga telegram chat_id qo'shish
            Lead.browse(existing).write({'telegram_chat_id': chat_id})
            return
 
        Lead.create({
            'name': f"Telegram: {contact.get('first_name', '')} {contact.get('last_name', '')}",
            'contact_name': f"{contact.get('first_name', '')} {contact.get('last_name', '')}",
            'phone': contact.get('phone_number'),
            'telegram_chat_id': chat_id,
            'mizon_source': 'telegram_bot',
        })
 
    def _log_message(self, chat_id, text, message):
        # Chat ID bo'yicha lead topish va chatter'ga yozish
        Lead = request.env['crm.lead'].sudo()
        lead = Lead.search([('telegram_chat_id', '=', chat_id)], limit=1)
        if lead:
            lead.message_post(
                body=f"Telegram xabari: {text}",
                message_type='comment',
                subtype_xmlid='mail.mt_note'
            )

import requests
from odoo import models, fields, api
from odoo.exceptions import UserError
 
class MizonTelegramService(models.AbstractModel):
    _name = 'mizon.telegram.service'
    _description = 'Telegram xabar yuborish servisi'
 
    @api.model
    def send_message(self, chat_id, text, parse_mode='HTML'):
        bot_token = self.env['ir.config_parameter'].sudo()\
            .get_param('mizon_crm.telegram_bot_token')
        if not bot_token:
            raise UserError('Telegram Bot token sozlanmagan!')
 
        url = f'https://api.telegram.org/bot{bot_token}/sendMessage'
        resp = requests.post(url, json={
            'chat_id': chat_id,
            'text': text,
            'parse_mode': parse_mode,
        }, timeout=10)
 
        if not resp.ok:
            raise UserError(f'Telegram xato: {resp.text}')
        return resp.json()

10.1 Odoo test tizimi

# mizon_crm/tests/test_mizon_lead.py
from odoo.tests.common import TransactionCase
 
class TestMizonLead(TransactionCase):
 
    def setUp(self):
        super().setUp()
        self.Lead = self.env['crm.lead']
        self.stage_new = self.env.ref('mizon_crm.stage_new')
 
    def test_lead_score_from_meta(self):
        """Meta Lead Ads manbasi uchun ball 30 bo'lishi kerak"""
        lead = self.Lead.create({
            'name': 'Test Lead',
            'mizon_source': 'meta_fb_ads',
        })
        lead._compute_lead_score()
        self.assertEqual(lead.lead_score, 30)
 
    def test_lead_score_with_phone(self):
        """Telefon qo'shilsa ball 10 ga oshishi kerak"""
        lead = self.Lead.create({
            'name': 'Test Lead',
            'mizon_source': 'referral',   # 40 ball
            'phone': '+998901234567',     # +10 ball
        })
        lead._compute_lead_score()
        self.assertEqual(lead.lead_score, 50)
 
    def test_duplicate_detection(self):
        """Bir xil telefonda duplikat aniqlanishi kerak"""
        self.Lead.create({'name': 'Lead 1', 'phone': '+998901234567'})
        dup_id = self.Lead.check_duplicate(phone='+998901234567')
        self.assertTrue(dup_id, 'Duplikat aniqlanmadi!')

11.1 Funksional tekshiruvlar
	•	Facebook Lead Ads formasidan yuborilgan lead 60 soniya ichida Odoo CRM'da "Yangi Lead" bosqichida ko'rinishi.
	•	Telegram bot'ga /start bosilganda menejerning Telegram ID'si CRM hisobi bilan bog'lanishi.
